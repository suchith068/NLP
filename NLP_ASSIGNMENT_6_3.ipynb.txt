        "emissions = hmm_tagger._outputs\n",
        "\n",
        "def top_emissions(tag, n=5):\n",
        "    tag_emissions = []\n",
        "    if tag in emissions:\n",
        "        for word in emissions[tag].samples():\n",
        "            prob = emissions[tag].prob(word)\n",
        "            tag_emissions.append((word, prob))\n",
        "    return sorted(\n",
        "        tag_emissions,\n",
        "        key=lambda x: x[1],\n",
        "        reverse=True\n",
        "    )[:n]\n",
        "\n",
